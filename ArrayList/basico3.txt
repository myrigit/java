import java.util.ArrayList;  // Importamos ArrayList para poder usarlo

// Declaraci√≥n del ArrayList y adici√≥n de valores
ArrayList<Integer> valores = new ArrayList<>();
valores.add(23);
valores.add(57);
valores.add(89);
valores.add(12);
valores.add(75);
valores.add(33);
valores.add(98);

// Contador para contar cu√°ntos valores son mayores que 50
int mayor = 0;

// Recorremos la lista con un for-each
// "num" ya es el valor de cada elemento en la lista, no un √≠ndice
for (int num : valores) {  
    if (num > 50) {  // Si el n√∫mero es mayor que 50, sumamos 1 al contador
        mayor++;
    }
}

// Imprimimos el total de n√∫meros mayores que 50 fuera del for-each
System.out.println(mayor);
